def account(token: nil)
  return if token.nil? && !Auth.authorized?.nil?

  Errors.error_handle(token) unless token.nil?

  response = HTTParty.get("https://api.genius.com/account?access_token=#{token_ext(token)}").body
  JSON.parse(response)
end
